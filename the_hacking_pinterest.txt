CREATE TABLE 'user' ('id' INTEGER PRIMARY KEY AUTOINCREMENT, 'name' TEXT);
CREATE TABLE 'pins' ('id' INTEGER PRIMARY KEY AUTOINCREMENT, 'url' TEXT, 'user_id' INTEGER, FOREIGN KEY (user_id) REFERENCES user (id));
CREATE TABLE 'commentaires' ('id' INTEGER PRIMARY KEY AUTOINCREMENT, 'user_id' INTEGER, 'pins_id' INTEGER, 'commentaire' TEXT, FOREIGN KEY (user_id) REFERENCES user (id), FOREIGN KEY(pins_id) REFERENCES pins (id) );
